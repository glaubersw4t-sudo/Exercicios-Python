# Inserindo Valor

valor = int(input('Digite o valor do produto desejado' ))
print(valor)

# Definindo opção de pagamento

opcao1 = '1. pix'
opcao2 = '2. credito'
opcao3 = '3. crediario'

print('--Forma de Pagamento--')
print(opcao1)
print(opcao2)
print(opcao3)

# Selecionando opção de pagamento

escolha = input('Digite a opção desejada (1-3): ')
if escolha == '1':
    resultado = valor - (valor * 0.20)
    print('Valor com desconto:',resultado)
elif escolha == '2':
    parcela = (int(input('Escolha o número de parcelas 1 - 10: ')))
    if parcela <= 2:
        print(f'Valor da parcela:{valor}/{parcela}, valor total: {valor}')
    elif 2 < parcela < 7:
        parcelamento = valor * 0.05 + valor
        print('Valor da parcela:', parcelamento / parcela, 'valor total:', parcelamento )
    elif 7 <= parcela < 11:
        parcelamento = valor * 0.14 + valor
        print('Valor da parcela:', parcelamento / parcela, 'valor total:', parcelamento)
    else:
        print('Opção inválida')
elif escolha == '3':
    num_parcelas = int(input('Escolha o número de parcelas 1 - 24: '))
    if num_parcelas <= 10:
        valor_total = valor * 1.09
        print('Valor da parcela:', valor_total / num_parcelas, 'Valor total:', valor_total)
    elif num_parcelas <= 20:
        valor_total = valor * 1.15
        print('Valor da parcela:', valor_total / num_parcelas, 'Valor total:', valor_total)
    elif num_parcelas == 24:
        valor_total = valor * 1.25
        print('Valor da parcela:', valor_total / num_parcelas, 'Valor total:', valor_total)
    else:
        print('Operação Inválida')
